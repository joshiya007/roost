Basic Template

Front-end

Start with this basic HTML template, copy the HTML below to begin working with a minimal Roost front-end page.

<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="utf-8">
        <meta http-equiv="X-UA-Compatible" content="IE=edge">
        <meta name="viewport" content="width=device-width, initial-scale=1">
        <title>Roost - Material Design Real Estate</title>

        <!-- CSS -->
        <link rel="stylesheet" href="css/app_1.min.css">
        <link rel="stylesheet" href="css/app_2.min.css">
    </head>

    <body>
        <header id="header">
            <!-- Header Content -->
        </header>

        <section class="section">
            <div class="container">
                <header class="section__title">
                    <h1>Duis mollisest non commodo luctus nisierat porttito</h1>
                    <small>Vestibulum id ligula porta felis euismod semper</small>
                </header>
            </div>
        </section>

        <footer id="footer">
            <!-- Footer Content -->
        </footer>

        <!-- Javascript -->

        <!-- jQuery -->
        <script src="vendors/bower_components/jquery/dist/jquery.min.js"></script>

        <!-- Bootstrap -->
        <script src="vendors/bower_components/bootstrap/dist/js/bootstrap.min.js"></script>
    </body>
</html>

Alternatively, you can make a copy of already included empty-page.html to start working on it.

Dashboard

Copy the HTML below to begin working with a minimal Roost Dashboard page.

<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="utf-8">
        <meta http-equiv="X-UA-Compatible" content="IE=edge">
        <meta name="viewport" content="width=device-width, initial-scale=1">
        <title>Roost - Material Design Real Estate</title>

        <!-- CSS -->
        <link rel="stylesheet" href="../css/app_1.min.css">
        <link rel="stylesheet" href="../css/app_2.min.css">
    </head>

    <body>
        <header id="header-alt">
            <!-- Header Content -->
        </header>

        <main id="main">
            <aside id="main__sidebar">
                <!-- Aside Content -->
            </aside>

            <section id="main__content">
                <div class="action-header-alt">
                    <div class="action-header__item action-header__item--search">
                        <form>
                            <input type="text" placeholder="Search...">
                        </form>
                    </div>
                </div>

                <div class="main__container">
                    <header class="main__title">
                        <h2>Morbi leorisus porta </h2>
                        <small>Cras mattis consectetur purus sit amet fermentum</small>
                    </header>
                </div>
            </section>

            <footer id="footer-alt">
                <!-- Footer Content -->
            </footer>
        </main>

        <!-- Javascript -->

        <!-- jQuery -->
        <script src="../vendors/bower_components/jquery/dist/jquery.min.js"></script>

        <!-- Bootstrap -->
        <script src="../vendors/bower_components/bootstrap/dist/js/bootstrap.min.js"></script>
    </body>
</html>

Alternatively, you can make a copy of already included /dashboard/empty-page.html to start working on it.
